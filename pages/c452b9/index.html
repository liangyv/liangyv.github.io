    # 安全的类型检测

    • typeof在某些浏览器返回的类型不一定准确，比如老版本的safari就认为正则是function
    • instanceof存在作用域的问题

    比较靠谱的方式是通过利用原生Object对象的toString()方法返回的字符串。当然前提是这个toString()方法没有被覆盖。

    function isArray(value){
	return Object.prototype.toString.call(value) == "[object Array]";
}

function isFunction(value){
	return Object.prototype.toString.call(value) == "[object Function]";
}

function isRegExp(value){
	return Object.prototype.toString.call(value) == "[object RegExp]";
}

//一些JS库判断JSON对象的常用方法
var isNativeJSON = window.JSON && Object.prototype.toString.call(JSON) == "[object JSON]";

    # 作用域安全的构造函数

    两种情况下作用域会被污染：

    1. 构造函数不用new，内部的this很容易因为调用环境不同而被污染
    2. 父类已经好好写了，但是子类继承父类的时候不按常理出牌也需要重新指向一下原型为父类来避免
    function Person(name, age){
	this.name = name;
	this.age = age;
}

var p1 = new Person(“jaybai”, 18)；//正确的方式
var p2 = Person(“jaybai”, 18)；//没有new，name和age会被挂载到window对象上

    所以可以改成这样来实现：

    function Person(name, age){
	if(this instanceof Person){
		this.name = name;
		this.age = age;	
	}else{
		return new Person(name, age);
	}
}

var p1 = new Person(“jaybai”, 18)；//正确的方式
var p2 = Person(“jaybai”, 18)；//也可以正常使用了

    如果子类骚操作还是可能被坑,因为Person的作用域安全的，调用Person.call的时候this对象并非是Person对象的实例，所以Person对象会返回一个new出来的新Person对象。Police中的this并没有被增长。

    function Person(name, age){
	if(this instanceof Person){
		this.name = name;
		this.age = age;	
	}else{
		return new Person(name, age);
	}
}

function Police(name, age, number){
	Person.call(this, name, age);
	this.number = number;
}

var p1 = new Police("jaybai", 18, 123456);
alert(p1.name); //undefined

    修改为：

    function Person(name, age){
	if(this instanceof Person){
		this.name = name;
		this.age = age;	
	}else{
		return new Person(name, age);
	}
}

function Police(name, age, number){
	Person.call(this, name, age);
	this.number = number;
}

Police.prototype = new Person();

var p1 = new Police("jaybai", 18, 123456);
alert(p1.name); //jaybai

    # 惰性载入函数

    惰性载入函数的意思是有大量分支逻辑的函数，但是这些分支只需要选择一次，后续都不需要再选择了，常用语浏览器特性的差异判断。有2两种方式来实现惰性函数。

    1. 在函数被调用时再处理函数。
    2. 在函数声明时就指定适当的函数。损失加载性能，但是能提高第一次执行性能。
    //比如创建浏览器XHR对象，需要做一堆判断，但是实际上只需要判断一次，后面都没有判断的必要了
function createXHR( ){
	if (typeof XMLHttpRequest != "undefined"){
    		return new XMLHttpRequest();
	} else if (typeof ActiveXObject != "undefined"){
		if (typeof arguments.callee.activeXString != "string"){
			var versions = ["MSXML2.XMLHttp.6.0", "MSXML2.XMLHttp.3.0", "MSXML2.XMLHttp"];   
			for (var i=0,len=versions.length; i < len; i++){
				try {
					new ActiveXObject(versions[i]);
					arguments.callee.activeXString = versions[i];
					break;
				} catch (ex){
					//skip
				}
			}
		}      
		return new ActiveXObject(arguments.callee.activeXString);
	} else {
		throw new Error("No XHR object available.");
	}
}

    所以，可以用惰性函数来修改这段代码，先用方法一，调用时重置一下函数实现就可以了

    function createXHR( ){
	if (typeof XMLHttpRequest != "undefined"){
		createXHR = function( ){
    			return new XMLHttpRequest();
		};
	} else if (typeof ActiveXObject != "undefined"){
		createXHR = function( ){
			if (typeof arguments.callee.activeXString != "string"){
				var versions = ["MSXML2.XMLHttp.6.0", "MSXML2.XMLHttp.3.0", "MSXML2.XMLHttp"];   
				for (var i=0,len=versions.length; i < len; i++){
					try {
						new ActiveXObject(versions[i]);
						arguments.callee.activeXString = versions[i];
						break;
					} catch (ex){
						//skip
					}
				}
			}        
			return new ActiveXObject(arguments.callee.activeXString);
		}
	} else {
		createXHR = function( ){
			throw new Error("No XHR object available.");
		}
	}
	return createXHR();
}

    用方法二来改造的话，就是函数声明的时候就决定好用哪个

    var createXHR = (function( ){
	if (typeof XMLHttpRequest != "undefined"){
    		return function( ){
    			return new XMLHttpRequest();
    		};
	} else if (typeof ActiveXObject != "undefined"){
		return function( ){
			if (typeof arguments.callee.activeXString != "string"){
				var versions = ["MSXML2.XMLHttp.6.0", "MSXML2.XMLHttp.3.0", "MSXML2.XMLHttp"];   
				for (var i=0,len=versions.length; i < len; i++){
					try {
						new ActiveXObject(versions[i]);
						arguments.callee.activeXString = versions[i];
						break;
					} catch (ex){
						//skip
					}
				}
			}      
			return new ActiveXObject(arguments.callee.activeXString);
		};

	}
})();

    # 函数绑定

    函数绑定其实就是绑定this的运行环境，常用于函数回调和事件处理程序。

    var user = {
	name: "jaybai",
	sayHi() {
		alert(`Hello, ${this.name}!`);
	}
};
setTimeout(user.sayHi, 1000); // Hello,!

    this的运行环境并不是我们希望的user对象内部，那么我们可以通过闭包来解决这个问题。

    var user = {
	name: "jaybai",
	sayHi() {
		alert(`Hello, ${this.name}!`);
	}
};
setTimeout(function( ){
		user.sayHi();
	}, 1000); //Hello, jaybai!

    或者我们可以写一个bind函数来实现：

    function bind(fn, context){
	return function( ){
		return fn.apply(context, arguments);
	};
}

var user = {
	name: "jaybai",
	sayHi() {
		alert(`Hello, ${this.name}!`);
	}
};

setTimeout(bind(user.sayHi, user), 1000); //Hello, jaybai!

    ES5之后也提供了原生的bind方法：

    var user = {
	name: "jaybai",
	sayHi() {
		alert(`Hello, ${this.name}!`);
	}
};

setTimeout(user.sayHi.bind(user), 1000); //Hello, jaybai!

    # 函数柯里化

    函数柯里化(function currying)听起来很高端，但是实现方式跟函数绑定是一样的，都是用闭包返回一些函数。区别在于，在函数调用的时候需要处理一些参数。

    // 普通的add函数
function add(x, y) {
    return x + y;
}

// Currying后
function curryingAdd(x) {
    return function (y) {
        return x + y;
    }
}

alert(add(1, 2));           // 3
alert(curryingAdd(1)(2));   // 3

    通用封装

    // 初步封装
var currying = function(fn) {
    // args 获取第一个方法内的全部参数
    var args = Array.prototype.slice.call(arguments, 1);
    return function( ) {
        // 将后面方法里的全部参数和args进行合并
        var newArgs = args.concat(Array.prototype.slice.call(arguments));
        // 把合并后的参数通过apply作为fn的参数并执行
        return fn.apply(this, newArgs);
    }
}

    多级参数柯里化封装

    // 支持多参数传递
function progressCurrying(fn, args) {

    var _this = this;
    var len = fn.length;
    var args = args || [];

    return function( ) {
        var _args = Array.prototype.slice.call(arguments);
        Array.prototype.push.apply(args, _args);

        // 如果参数个数小于最初的fn.length，则递归调用，继续收集参数
        if (_args.length < len) {
            return progressCurrying.call(_this, fn, _args);
        }

        // 参数收集完毕，则执行fn
        return fn.apply(this, _args);
    }
}

    # 防篡改对象

    ES5定义了防篡改对象（tamper-proof object）

    类型 方法 描述 isExtensible() isSeaked() isFrozen()
    可修改对象 默认对象就是这种 可以随便新增、修改、删除成员 true false false
    不可扩展对象 Object.preventExtensions() 不能新增对象属性，非严格模式静默失败，严格模式报错。但是可以修改和删除已有成员 false false false
    密封的对象 Object.seal() 不能新增、删除对象属性，非严格模式静默失败，严格模式报错。但是可以修改属性值 false true false
    冻结的对象 Object.freeze() 不能新增、修改、删除对象属性，非严格模式静默失败，严格模式报错。但是可以通过[[Set]]函数修改成员属性值 false true true
    /* * 本案例由wendyye提供部分思路，解释一下冻结后何种情况可以通过[[set]]函数修改 * 然后可以看出来冻结的只有一层对象，其实是浅冻结 */
var codeName = "jaybai"; //外部对象无法被冻结
var coder = {
	age : 18, //内部对象都被冻结住了
	
	pc : {
		os : "macOS" //内部对象的属性不会被冻结
	},
	
	set name(x){
		codeName = x;
	},
	get name(){
		return codeName;
	}
};

console.log('修改前 name:', coder.name); // jaybai
console.log('修改前 age:', coder.age); // 18
console.log('修改前 pc.os:', coder.pc.os); // macOS

Object.freeze(coder); //冻结

coder.name = "baijie";
coder.age = 28;
coder.pc.os = "Windows";
console.log('修改后 name:', coder.name); // baijie
console.log('修改后 age:', coder.age); // 18
console.log('修改后 pc.os:', coder.pc.os); // Windows

delete coder.name;
delete coder.age;
delete coder.pc.os;
console.log('删除后 name:', coder.name); // baijie
console.log('删除后 age:', coder.age); // 18
console.log('删除后 pc.os:', coder.pc.os); // undefined

    浅冻结只能冻结当前对象、数组等内部第一层，如果第一层中，还嵌套着对象、数组等，那么这种更深的层级，浅冻结是无法干预到的；对象中的对象，其实真正的值，又放在堆内存当中了。如果要深冻结，就需要通过递归去处理了。

    /* * 深度冻结 */
function deepFreeze(o) {
	var prop, propKey
	Object.freeze(o) // 首先冻结第一层对象
	for (propKey in o) {
		prop = o[propKey];
		if (!o.hasOwnProperty(propKey) || !(typeof prop === "object") || Object.isFrozen(prop)) {
			continue;// 跳过原型链上的属性、基本类型和已冻结的对象.
		}
		deepFreeze(prop);//递归调用.
	}
}

var coder = {
	pc : {
		os : "macOS" 
	}
};

console.log('修改前 pc.os:', coder.pc.os); // macOS
deepFreeze(coder); //冻结

coder.pc.os = "Windows";
console.log('修改后 pc.os:', coder.pc.os); // macOS

delete coder.pc.os;
console.log('删除后 pc.os:', coder.pc.os); // macOS

    # 高级定时器

    # 重复的定时器

    JavaScript是只有一个单进程执行代码。所以不管是setTimeout()，还是setInterval()都是不能保证准时执行的。JS中除了执行主进程，还有一个待执行队列的概念，任何代码的执行都是被加入到这个待执行队列中，等待主进程空闲的时候再去执行的。所以定时器的概念不是定义何时执行，而是定义代码何时被加入到这个待执行队列中。

    setInterval()创建的重复定时器，虽然比较聪明的帮我们解决了，如果加入一个新定时器实例的时候已经存在一个定时器就不在加入，会跳过。但是这里还存在另外一个问题，如果定时器内部逻辑代码的执行时间比较长，每次执行完都有一个新的定时器实例在等着执行下一次，就会导致定时器代码持续不断的在执行。

    所以比较好的方式，是结合setTimeout()和setInterval()的思想：

    setTimeout(function( ){
	//这里先执行定时处理的业务代码
	//interval就是你具体想要间隔的时间值
	setTimeout(arguments.callee, interval);
}, interval)；

    # Yielding Processes

    脚本长时间运行会导致浏览器弹窗询问是否继续执行。

    长时间运行的原因可能有2种：

    1. 过长、过深的嵌套函数调用
    2. 大量处理的循环

    对于大量处理的循环问题，先问两个问题：

    1. 该处理是否必须是同步完成的
    2. 数据是否必须按顺序完成

    如果答案是否，那么就可以用定时器分割这个循环。这个技术叫做数组分块（array chucking）:

    function chuck(array, process, context){
	setTimeout(function( ){
		//取出并允许一条数据
		var item = array.shift(); //array的本质就是一个待办事项列表
		process(item);
		
		//如果还有待处理的数据，则设置一个新定时器处理
		if(array.length > 0){
			setTimeout(arguments.callee, 100);
		}		
	}, 100)；
}

	claerTimeout(method.tId);
