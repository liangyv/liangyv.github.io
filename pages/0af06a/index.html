    第7章函数表达式

    定义函数的2种方式:

    1. 函数声明
    2. 函数表达式
    //函数声明
function a(arg0, arg1){
	//函数体
}

//函数表达式
var b = function(arg0, arg1){
	//函数体
};

    # 递归

    如前所述，递归的时候注意用arguments.callee来代替递归的函数名。用来解耦函数名。唯一有个问题就是在严格模式下不能访问arguments.callee这个属性，会报错了。可以使用明明函数表达式来解决。

    var factorial = (function f(x){
	if(x <= 1){
		return 1;
	}else{
		return x * f(x-1);
	}
});

    # 闭包

    闭包是指有权访问另一个函数作用域中变量的函数。创建闭包的常用方式，就是在一个函数内部创建另外一个函数。

    function createComparisonFunction(p){
	
	return function(o1, o2){
		var v1 = o1[p]; //这里访问了外部变量p
		var v2 = o2[p]; //这里访问了外部变量p
		
		if(v1 < v2){
			return -1;
		}else if(v1 > v2){
			return 1;
		}else{
			return 0;
		}
	};
}

var compare = createComparisonFunction("name");
var result = compare({name: "Nicholas"}, {name: "Greg"});

    闭包会携带包含它的函数作用域，因此会占用比较多的内存。必要时才使用！！！

    闭包中的变量

    //这个函数会返回一个函数数组，但是每个函数都是返回的10。因为每个函数的作用域链中都保存这aFunc()函数的活动对象，都是引用的同一个变量i.
function aFunc( ){
	var r = new Array();
	
	for(var i=0; i<10; i++){
		r[i] = function( ){
			return i;
		};	
	}
	return r;
}
//上面那个问题，恶意通过另一个匿名函数解决
function aFunc( ){
	var r = new Array();
	
	for(var i=0; i<10; i++){
		r[i] = function(n){
			return function( ){
				return n;
			};
		}(i);	
	}
	return r;
}

    关于this对象

    this对象是在运行时基于函数的执行环境绑定的。不过，匿名函数的执行环节具有全局性。this通常会指向window。但也不全是这样。

    var name = "window";

var o = {
	name : "o",
	getName : function( ){
		return function( ){
			return this.name;
		};
	}
};

alert(o.getName()()); //windos ——在非严格模式下

    函数被调用时，有2个特殊变量this和arguments。内部函数索索变量是，只会搜索到其活动对象为止，因为永远不可能直接访问外部函数中的这2个变量。如果要访问，只能把他们保存到闭包能访问到的变量里。

    var name = "window";

var o = {
	name : "o",
	getName : function( ){
		var that = this; //把this放到一个闭包可以访问的变量的位置
		return function( ){
			return that.name;
		};
	}
};

alert(o.getName()()); //o
var name = "window";

var o = {
	name : "o",
	getName : function( ){
		return this.name;
	}
};


console.log(o.getName()); //o
console.log((o.getName)()); //o
//复制语句相当于把一个函数复制给了一个函数，this的值得不到维持，变成了全局环境
console.log((o.getName = o.getName)()); //window

    内存泄漏

    如前所述，IE9之前JS对象和COM对象的不同垃圾回收策略导致了如果闭包中保存着一个HTML元素，就无法销毁。

    function a( ){
	var e = document.getElementById("aaa");
	e.onclick = function( ){
		alert(e.id); //匿名函数内部持有对e的引用，至少是1，e无法被销毁
	};
}

    可以这么改写

    function a( ){
	var e = document.getElementById("aaa");
	var id = e.id;
	e.onclick = function( ){
		alert(id); 
	};
	e = null;
}

    模仿块级作用域

    • JS没有块级作用域的概念，比如之前的for循环中i的例子
    • JS的作用域是包含在函数中（或者说环境上下文中的）
    • JS不会对多次定义变量报错，会忽略后面的定义，但是会执行后面的初始化方法
    //匿名函数模仿块级作用域
(function( ){
	//这里是块级的作用域
})();

    小括号扩起来：是把函数声明转换成函数表达式。

    function a(x){
	(function( ){
		for(var i=0, i < x, i++){
			alert(i);
		}
	})();
	alert(i); //不会像之前打印出x的值了，会报错
}

    # 私有变量

    严格来讲JS中没有私有成员的概念，所有对象属性都是公有的。只有一个私有变量的概念，在函数中定义的变量、局部变量、函数内部定义的其他函数。

    有权访问私有变量和私有函数的共有方法称为特权方法（privaileged method），有2种创建特权方法的方式

    1. 构造函数中定义
    2. 利用私有和特权成员
    //构造函数中定义特权方法
function MyObject( ){
	//私有变量
	var a = 1;
	
	//私有函数
	function b( ){
		return false;
	}
	
	//特权方法
	this.c = function( ){
		a++;
		return b();
	};
}
//利用私有和特权成员
function Coder(name){
	//特权方法
	this.getName = function( ){
		return name;
	};
	//特权方法
	this.setName = function(v){
		name = v;
	};
}

var coder = new Coder("jaybai");
alert(coder.getName()); //jaybai
coder.setName("baijie");
alert(coder.getName()); //baijie

    静态私有变量

    //这个方式会导致实例共享私有变量和函数，需要特别注意
(function( ){
	//私有变量
	var a = 1;
	
	//私有函数
	function b( ){
		return false;
	}
	
	//构造函数
	//用函数表达式。（函数声明的方式创建的是局部函数，比如上面的function b()）
	//同样不能使用var变量，道理同上
	MyObject = function( ){
	};
	
	//公有/特权方法
	MyObject.prototype.c = function( ){
		a++;
		return b();
	};
})();

    模块模式（module pattern）

    为单例创建私有变量和特权方法

    var singleton = function( ){
	//私有变量
	var a = 1;
	
	//私有函数
	function b( ){
		return false;
	}
	
	//公有/特权方法
	return {
		pa : true;
		pb : function( ){
			a++;
			return b();
		}
	};
};

    增强的模块模式

    //特殊类型
function CustomType( ){
	//具体代码
};

var singleton = function( ){
	//私有变量
	var a = 1;
	
	//私有函数
	function b( ){
		return false;
	}
	
	//创建对象
	var o = new CustomType();
	
	//添加公有/特权方法
	o.pa = true;
	o.pb = function( ){
		a++;
		return b();
	};
	
	return o;
};
